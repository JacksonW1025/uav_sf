   300	template <typename T>
   301	void quaternion_multiplication(T q1[4], T q2[4], T q_res[4])
   302	{
   303		q_res[0] = q1[0] * q2[0] - q1[1] * q2[1] - q1[2] * q2[2] - q1[3] * q2[3];
   304		q_res[1] = q1[0] * q2[1] + q1[1] * q2[0] + q1[2] * q2[3] - q1[3] * q2[2];
   305		q_res[2] = q1[0] * q2[2] - q1[1] * q2[3] + q1[2] * q2[0] + q1[3] * q2[1];
   306		q_res[3] = q1[0] * q2[3] + q1[1] * q2[2] - q1[2] * q2[1] + q1[3] * q2[0];
   307	}
   308	template <typename T>
   309	void quaternion_conjugate(T q[4], T q_res[4])
   310	{
   311		q_res[0] = +q[0];
   312		q_res[1] = -q[1];
   313		q_res[2] = -q[2];
   314		q_res[3] = -q[3];
   315	}
   316	template <typename T>
   317	void quaternion_to_rotation_matrix(T q[4], T R[9])
   318	{
   319		// row-major
   320		T qw = q[0];
   321		T qx = q[1];
   322		T qy = q[2];
   323		T qz = q[3];
   324	
   325		R[0] = 1 - 2 * qy * qy - 2 * qz * qz;
   326		R[1] =     2 * qx * qy - 2 * qw * qz;
   327		R[2] =     2 * qx * qz + 2 * qw * qy;
   328		R[3] =     2 * qx * qy + 2 * qw * qz;
   329		R[4] = 1 - 2 * qx * qx - 2 * qz * qz;
   330		R[5] =     2 * qy * qz - 2 * qw * qx;
   331		R[6] =     2 * qx * qz - 2 * qw * qy;
   332		R[7] =     2 * qy * qz + 2 * qw * qx;
   333		R[8] = 1 - 2 * qx * qx - 2 * qy * qy;
   334	}

   336	template <typename T>
   337	void rotate_vector(T R[9], T v[3], T v_rotated[3])
   338	{
   339		v_rotated[0] = R[0] * v[0] + R[1] * v[1] + R[2] * v[2];
   340		v_rotated[1] = R[3] * v[0] + R[4] * v[1] + R[5] * v[2];
   341		v_rotated[2] = R[6] * v[0] + R[7] * v[1] + R[8] * v[2];
   342	}

   344	void Raptor::observe(rl_tools::inference::applications::l2f::Observation<EXECUTOR_SPEC> &observation)
   345	{
   346		// converting from FRD to FLU
   347		T Rt_inv[9];
   348	
   349		{
   350			// Orientation
   351			// FRD to FLU
   352			T q_target[4];
   353			q_target[0] = cosf(0.5f * _trajectory_setpoint.yaw); // minus because the setpoint yaw is in NED
   354			q_target[1] = 0;
   355			q_target[2] = 0;
   356			q_target[3] = sinf(0.5f * _trajectory_setpoint.yaw);
   357	
   358			T qt[4], qtc[4], qr[4];
   359			qt[0] = +q_target[0]; // conjugate to build the difference between setpoint and current
   360			qt[1] = +q_target[1];
   361			qt[2] = -q_target[2];
   362			qt[3] = -q_target[3];
   363			quaternion_conjugate(qt, qtc);
   364			quaternion_to_rotation_matrix(qtc, Rt_inv);
   365	
   366			qr[0] = +_vehicle_attitude.q[0];
   367			qr[1] = +_vehicle_attitude.q[1];
   368			qr[2] = -_vehicle_attitude.q[2];
   369			qr[3] = -_vehicle_attitude.q[3];
   370			// qr = qt * qd
   371			// qd = qt' * qr
   372			T qd[4];
   373			quaternion_multiplication(qtc, qr, qd);
   374	
   375			observation.orientation[0] = qd[0];
   376			observation.orientation[1] = qd[1];
   377			observation.orientation[2] = qd[2];
   378			observation.orientation[3] = qd[3];
   379		}
   380	
   381		{
   382			// Position
   383			T p[3], pt[3]; // FLU
   384			p[0] = +(position[0] - _trajectory_setpoint.position[0]);
   385			p[1] = -(position[1] - _trajectory_setpoint.position[1]);
   386			p[2] = -(position[2] - _trajectory_setpoint.position[2]);
   387			rotate_vector(Rt_inv, p, pt); // The position and velocity error are in the target frame
   388			observation.position[0] = clip(pt[0], max_position_error, -max_position_error);
   389			observation.position[1] = clip(pt[1], max_position_error, -max_position_error);
   390			observation.position[2] = clip(pt[2], max_position_error, -max_position_error);
   391		}
   392		{
   393			// Linear Velocity
   394			T v[3], vt[3];
   395			v[0] = +(linear_velocity[0] - _trajectory_setpoint.velocity[0]);
   396			v[1] = -(linear_velocity[1] - _trajectory_setpoint.velocity[1]);
   397			v[2] = -(linear_velocity[2] - _trajectory_setpoint.velocity[2]);
   398			rotate_vector(Rt_inv, v, vt);
   399			observation.linear_velocity[0] = clip(vt[0], max_velocity_error, -max_velocity_error);
   400			observation.linear_velocity[1] = clip(vt[1], max_velocity_error, -max_velocity_error);
   401			observation.linear_velocity[2] = clip(vt[2], max_velocity_error, -max_velocity_error);
   402		}
   403		{
   404			// Angular Velocity
   405			observation.angular_velocity[0] = +_vehicle_angular_velocity.xyz[0];
   406			observation.angular_velocity[1] = -_vehicle_angular_velocity.xyz[1];
   407			observation.angular_velocity[2] = -_vehicle_angular_velocity.xyz[2];
   408		}
   409	
   410		for (TI action_i = 0; action_i < EXECUTOR_CONFIG::OUTPUT_DIM; action_i++) {
   411			observation.previous_action[action_i] = this->previous_action[action_i];
   412		}
   413	}

   416	void Raptor::updateArmingCheckReply()
   417	{
   418		if (flightmode_state == FlightModeState::CONFIGURED) {
   419			if (_arming_check_request_sub.updated()) {
   420				arming_check_request_s arming_check_request;
   421				_arming_check_request_sub.copy(&arming_check_request);
   422				arming_check_reply_s arming_check_reply;
   423				arming_check_reply.timestamp = hrt_absolute_time();
   424				arming_check_reply.request_id = arming_check_request.request_id;
   425				arming_check_reply.registration_id = ext_component_arming_check_id;
   426				arming_check_reply.health_component_index = arming_check_reply.HEALTH_COMPONENT_INDEX_NONE;
   427				arming_check_reply.num_events = 0;
   428				arming_check_reply.can_arm_and_run = can_arm;
   429				arming_check_reply.mode_req_angular_velocity = true;
   430				arming_check_reply.mode_req_local_position = true;
   431				arming_check_reply.mode_req_attitude = true;
   432				arming_check_reply.mode_req_local_alt = true;
   433				arming_check_reply.mode_req_home_position = false;
   434				arming_check_reply.mode_req_mission = false;
   435				arming_check_reply.mode_req_global_position = false;
   436				arming_check_reply.mode_req_prevent_arming = false;
   437				arming_check_reply.mode_req_manual_control = false;
   438				_arming_check_reply_pub.publish(arming_check_reply);
   439			}
   440		}
   441	}

   444	void Raptor::Run()
   445	{
   446		if (should_exit()) {
   447			_vehicle_angular_velocity_sub.unregisterCallback();
   448	
   449			if (flightmode_state >= FlightModeState::REGISTERED) {
   450				unregister_ext_component_s unregister_ext_component{};
   451				unregister_ext_component.timestamp = hrt_absolute_time();
   452				strncpy(unregister_ext_component.name, "RAPTOR", sizeof(unregister_ext_component.name) - 1);
   453				unregister_ext_component.arming_check_id = ext_component_arming_check_id;
   454				unregister_ext_component.mode_id = ext_component_mode_id;
   455				unregister_ext_component.mode_executor_id = -1;
   456				_unregister_ext_component_pub.publish(unregister_ext_component);
   457			}
   458	
   459			ScheduleClear();
   460			exit_and_cleanup(desc);
   461			return;
   462		}
   463	
   464		register_ext_component_reply_s register_ext_component_reply;
   465	
   466		if (_register_ext_component_reply_sub.update(&register_ext_component_reply)) {
   467			if (register_ext_component_reply.request_id == Raptor::EXT_COMPONENT_REQUEST_ID && register_ext_component_reply.success) {
   468				ext_component_arming_check_id = register_ext_component_reply.arming_check_id;
   469				ext_component_mode_id = register_ext_component_reply.mode_id;
   470				flightmode_state = FlightModeState::REGISTERED;
   471				PX4_INFO("Raptor mode registration successful, arming_check_id: %d, mode_id: %d", ext_component_arming_check_id, ext_component_mode_id);
   472			}
   473		}
   474	
   475		if (flightmode_state == FlightModeState::REGISTERED) {
   476			vehicle_control_mode_s config_control_setpoints{};
   477			config_control_setpoints.timestamp = hrt_absolute_time();
   478			config_control_setpoints.source_id = ext_component_mode_id;

   512			for (TI i = 0; i < (trajectory_setpoint_dts_full ? NUM_TRAJECTORY_SETPOINT_DTS : trajectory_setpoint_dt_index); i++) {
   513				TI index = trajectory_setpoint_dts_full ? i : trajectory_setpoint_dt_index - 1 - i;

   526			status.trajectory_setpoint_dt_mean = trajectory_setpoint_dt_mean;
   527			status.trajectory_setpoint_dt_max = trajectory_setpoint_dt_max;
   528			status.trajectory_setpoint_dt_max_since_activation = trajectory_setpoint_dt_max_since_reset;
   529		}
   530	
   531		status.subscription_update_vehicle_status = _vehicle_status_sub.update(&_vehicle_status);
   532	
   533		if (status.subscription_update_vehicle_status) {
   534			timestamp_last_vehicle_status = current_time;
   535			timestamp_last_vehicle_status_set = true;
   536		}
   537	
   538		bool next_active = timestamp_last_vehicle_status_set && _vehicle_status.nav_state == ext_component_mode_id;
   539	
   540		if (!previous_active && next_active) {
   541			this->reset();
   542			PX4_INFO("Resetting Inference Executor (Recurrent State)");
   543	
   544		} else {
   545			if (previous_active && !next_active) {
   546				PX4_INFO("inactive");
   547			}

   550		bool angular_velocity_update = false;

   557		}
   558

   581		bool use_external_reference = !use_internal_reference;

   586				PX4_ISFINITE(temp_trajectory_setpoint.position[0]) &&

   588				PX4_ISFINITE(temp_trajectory_setpoint.position[2]) &&

   600						if (next_active && !trajectory_setpoint_dts_full) {
   601							PX4_INFO("trajectory_setpoint_dts_full");
   602						}
   603	
   604						trajectory_setpoint_dts_full = true;
   605						trajectory_setpoint_dt_index = 0;
   606					}
   607				}
   608	
   609				timestamp_last_trajectory_setpoint_set = true;
   610				status.timestamp_last_trajectory_setpoint = current_time;
   611				timestamp_last_trajectory_setpoint = current_time;

   614			} else {
   615				trajectory_setpoint_invalid_count++;

   617				if (next_active && trajectory_setpoint_invalid_count % TRAJECTORY_SETPOINT_INVALID_COUNT_WARNING_INTERVAL == 0) {
   618					PX4_WARN("trajectory_setpoint invalid, count: %d", (int)trajectory_setpoint_invalid_count);
   619				}
   620			}
   621		}

   623		constexpr bool PUBLISH_NON_COMPLETE_STATUS = true;
   624	
   625		if (!angular_velocity_update) {
   626			status.exit_reason = raptor_status_s::EXIT_REASON_NO_ANGULAR_VELOCITY_UPDATE;
   627

   629				_raptor_status_pub.publish(status);
   630			}
   631

   635	
   636		if (!timestamp_last_angular_velocity_set || !timestamp_last_local_position_set || !timestamp_last_attitude_set) {
   637			status.exit_reason = raptor_status_s::EXIT_REASON_NOT_ALL_OBSERVATIONS_SET;

   640			status.vehicle_attitude_stale = !timestamp_last_attitude_set;

   758			}
   759

   761				PX4_ERR("attitude timeout");
   762				timeout_message_sent = true;
   763			}
   764	
   765			can_arm = false;
   766			updateArmingCheckReply();
   767			return;
   768		}
   769	
   770		timeout_message_sent = false;
   771	
   772		// is ready to control at this point
   773		can_arm = true;
   774		updateArmingCheckReply();

   776		if (!timestamp_last_trajectory_setpoint_set || use_internal_reference
   777		    || (current_time - timestamp_last_trajectory_setpoint) > TRAJECTORY_SETPOINT_TIMEOUT) {
   778			status.trajectory_setpoint_stale = true;
   779	
   780			if (!previous_trajectory_setpoint_stale || (!previous_active && next_active)) {
   781				_trajectory_setpoint.position[0] = position[0];
   782				_trajectory_setpoint.position[1] = position[1];
   783				_trajectory_setpoint.position[2] = position[2];

   785				_trajectory_setpoint.yaw = atan2f(2.0f * (q[1] * q[2] + q[0] * q[3]), 1.0f - 2.0f * (q[2] * q[2] + q[3] * q[3]));

   787				_trajectory_setpoint.velocity[1] = 0;
   788				_trajectory_setpoint.velocity[2] = 0;
   789				_trajectory_setpoint.yawspeed = 0;
   790

   792					PX4_WARN("trajectory_setpoint turned stale at: %f %f %f, yaw: %f %llu / %llu us", (double)position[0], (double)position[1],
   793						 (double)position[2],
   794						 (double)_trajectory_setpoint.yaw, (unsigned long long)(current_time - timestamp_last_trajectory_setpoint),
   795						 (unsigned long long)(TRAJECTORY_SETPOINT_TIMEOUT));
   796	
   797				} else {
   798					PX4_WARN("trajectory_setpoint reset due to activation at: %f %f %f, yaw: %f", (double)position[0], (double)position[1], (double)position[2],
   799						 (double)_trajectory_setpoint.yaw);
   800				}
   801			}
   802	
   803			previous_trajectory_setpoint_stale = true;
   804	
   805		} else {
   806			if (previous_trajectory_setpoint_stale) {
   807				PX4_WARN("trajectory_setpoint turned non-stale at: %f %f %f", (double)position[0], (double)position[1], (double)position[2]);
   808				previous_trajectory_setpoint_stale = false;
   809			}
   810	
   811			status.trajectory_setpoint_stale = false;
   812		}

   820		observe(observation);
   821		hrt_abstime nanoseconds = current_time * 1000;
   822		auto executor_status = rl_tools::control(device, executor, nanoseconds, policy, observation, action, rng);
   823	
   824		if (!executor_status.OK) {
   825			if (executor_status.TIMESTAMP_INVALID) {
   826				PX4_ERR("RLtools executor error: Timestamp invalid");
   827			}
   828	
   829			if (executor_status.LAST_CONTROL_TIMESTAMP_GREATER_THAN_LAST_OBSERVATION_TIMESTAMP) {

   832			}
   833		}
   834	
   835		if (executor_status.source != decltype(executor_status.source)::CONTROL) {
   836			// status.exit_reason = raptor_status_s::EXIT_REASON_EXECUTOR_STATUS_SOURCE_NOT_CONTROL;
   837			// if constexpr(PUBLISH_NON_COMPLETE_STATUS){
   838			// 	_raptor_status_pub.publish(status);
   839			// }
   840			// Exit early if it is not time to control

   851		input_msg.active = status.active;
   852		static_assert(raptor_input_s::ACTION_DIM == EXECUTOR_CONFIG::OUTPUT_DIM);
   853		input_msg.timestamp = current_time;
   854		input_msg.timestamp_sample = _vehicle_angular_velocity.timestamp_sample;
   855	
   856		for (TI dim_i = 0; dim_i < 3; dim_i++) {
   857			input_msg.position[dim_i] = observation.position[dim_i];
   858			input_msg.orientation[dim_i] = observation.orientation[dim_i];
   859			input_msg.linear_velocity[dim_i] = observation.linear_velocity[dim_i];
   860			input_msg.angular_velocity[dim_i] = observation.angular_velocity[dim_i];
   861		}
   862	
   863		input_msg.orientation[3] = observation.orientation[3];
   864	
   865		for (TI dim_i = 0; dim_i < EXECUTOR_CONFIG::OUTPUT_DIM; dim_i++) {
   866			input_msg.previous_action[dim_i] = observation.previous_action[dim_i];
   867		}
   868	
   869		_raptor_input_pub.publish(input_msg);
   870		_raptor_status_pub.publish(status);
   871	
   872		actuator_motors_s actuator_motors{};
   873		actuator_motors.timestamp = hrt_absolute_time();
   874		actuator_motors.timestamp_sample = _vehicle_angular_velocity.timestamp_sample;
   875	
   876		for (TI action_i = 0; action_i < actuator_motors_s::NUM_CONTROLS; action_i++) {
   877			if (action_i < EXECUTOR_CONFIG::OUTPUT_DIM) {
   878				T value = action.action[action_i];
   879				this->previous_action[action_i] = value;
   880				value = (value + 1) / 2;
   881				constexpr T training_min = 0;
   882				constexpr T training_max = 1.0;
   883				T scaled_value = (training_max - training_min) * value + training_min;
   884				actuator_motors.control[action_i] = scaled_value;
   885	
   886			} else {
   887				actuator_motors.control[action_i] = NAN;
   888			}
   889		}
   890	
   891		if constexpr(Raptor::REMAP_FROM_CRAZYFLIE) {
   892			actuator_motors_s temp = actuator_motors;
   893			temp.control[0] = actuator_motors.control[0];
   894			temp.control[1] = actuator_motors.control[2];
   895			temp.control[2] = actuator_motors.control[3];
   896			temp.control[3] = actuator_motors.control[1];
   897			actuator_motors = temp;
   898		}

    73	        using TI = typename SPEC::TI;
    74	        if(executor.steps_since_original_control_step == 0){
    75	            for(TI action_i = 0; action_i < SPEC::OUTPUT_DIM; action_i++){
    76	                executor.action_history[0][action_i] = observation.previous_action[action_i];
    77	            }
    78	        }
    79	        else{
    80	            for(TI action_i = 0; action_i < SPEC::OUTPUT_DIM; action_i++){
    81	                executor.action_history[0][action_i] = (executor.action_history[0][action_i] * executor.steps_since_original_control_step + observation.previous_action[action_i]) / (executor.steps_since_original_control_step + 1);
    82	            }
    83	        }
    84	        inference::applications::l2f::observe(device, executor, observation, executor.input);
    85	        auto status = control(device, executor.executor, nanoseconds, policy, executor.input, executor.output, rng);
    86	        for (TI output_i=0; output_i < SPEC::OUTPUT_DIM; output_i++){
    87	            action.action[output_i] = get(device, executor.output, 0, output_i);
    88	        }
    89	
    90	        executor.steps_since_original_control_step++; // gets overwritten with 0 in the case of an original control step
    91	        if(status.source == decltype(status.source)::CONTROL){
    92	            if(status.step_type == decltype(status.step_type)::NATIVE){
    93	                // step action history
    94	                static_assert(SPEC::ACTION_HISTORY_LENGTH >= 1);
    95	                for(TI step_i = SPEC::ACTION_HISTORY_LENGTH-1; step_i > 0; step_i--){
    96	                    for(TI action_i = 0; action_i < SPEC::OUTPUT_DIM; action_i++){
    97	                        executor.action_history[step_i][action_i] = executor.action_history[step_i-1][action_i];
    98	                    }
    99	                }
   100	                executor.steps_since_original_control_step = 0;
   101	            }
   102	        }
   103	        return status;
   104	    }
